# Function which adds ogre next plugin
# If use_ogre2_definitions is true then old ogre2 (not ogre_next) definitions
# will be generated
function(gz_add_ogre_next_plugin use_ogre2_definitions)

  # Collect source files into the "sources" variable and unit test files into the
  # "gtest_sources" variable.
  gz_get_libsources_and_unittests(sources gtest_sources)

  if (MSVC)
    # Warning #4251 is the "dll-interface" warning that tells you when types used
    # by a class are not being exported. These generated source files have private
    # members that don't get exported, so they trigger this warning. However, the
    # warning is not important since those members do not need to be interfaced
    # with.
    set_source_files_properties(${sources} ${gtest_sources} COMPILE_FLAGS "/wd4251")
    set_source_files_properties(${sources} ${gtest_sources} COMPILE_FLAGS "/D_SILENCE_STDEXT_HASH_DEPRECATION_WARNINGS")
  endif()

  set(engine_name "ogre_next")
  if (use_ogre2_definitions)
    set(engine_name "ogre2")
  endif()

  gz_add_component(${engine_name}
    SOURCES ${sources}
    GET_TARGET_NAME ogre_next_target
  )

  if (use_ogre2_definitions)
    # Generate files needed by gz-rendering-ogre2 but not generated by
    # gz_add_component, including:
    # - gz/rendering/ogre2/Export.hh
    # - gz/rendering/ogre2/detail/Export.hh
    set(include_dir "gz/rendering/ogre2")
    set(export_base "GZ_RENDERING_OGRE2")
    set(lib_name "gz-rendering-ogre2")
    configure_file(
      "${GZ_CMAKE_DIR}/Export.hh.in"
      "${CMAKE_BINARY_DIR}/include/${include_dir}/Export.hh")
    install(
      FILES "${CMAKE_BINARY_DIR}/include/${include_dir}/Export.hh"
      DESTINATION "${GZ_INCLUDE_INSTALL_DIR_FULL}/ogre2/"
      COMPONENT headers)
    include(GenerateExportHeader)
    generate_export_header("gz-rendering-ogre2"
      BASE_NAME ${export_base}
      EXPORT_FILE_NAME
          "${PROJECT_BINARY_DIR}/include/gz/rendering/ogre2/detail/Export.hh"
      EXPORT_MACRO_NAME DETAIL_${export_base}_VISIBLE
      NO_EXPORT_MACRO_NAME DETAIL_${export_base}_HIDDEN
      DEPRECATED_MACRO_NAME GZ_DEPRECATED_ALL_VERSIONS)
    unset(lib_name)
    unset(export_base)
    unset(include_dir)
    # Include ogre_next definitions
    target_include_directories(${ogre_next_target}
      PUBLIC
        "$<BUILD_INTERFACE:${PROJECT_SOURCE_DIR}/ogre_next/include>")
    target_include_directories(${ogre_next_target}
      PUBLIC
        "$<BUILD_INTERFACE:${PROJECT_BINARY_DIR}/ogre_next/include>")
  endif()

  set(OGRE_NEXT_RESOURCE_PATH_STR "${OGRE_NEXT_RESOURCE_PATH}")
  if (NOT OGRE_NEXT_RESOURCE_PATH_STR)
    set(OGRE_NEXT_RESOURCE_PATH_STR "${OGRE2_RESOURCE_PATH}")
  endif()
  # On non-Windows, we need to convert the CMake list delimited (;) to the
  # list delimiter used in list of paths in code (:)
  # On Windows, the list delimiter in code is already ;, not need to change it to :
  if(NOT WIN32)
    string(REPLACE ";" ":" OGRE_NEXT_RESOURCE_PATH_STR
      "${OGRE_NEXT_RESOURCE_PATH_STR}"
    )
  endif()

  set_property(
    SOURCE OgreNextRenderEngine.cc
    PROPERTY COMPILE_DEFINITIONS
    OGRE_NEXT_RESOURCE_PATH="${OGRE_NEXT_RESOURCE_PATH_STR}"
    OGRE_NEXT_VERSION="${OGRE_NEXT_VERSION}"
  )

  set(ogre_next_backward_compatible_suffix _NEXT)
  if (NOT GZ_RENDERING_HAVE_OGRE_NEXT)
    set(ogre_next_backward_compatible_suffix 2)
  endif()
  target_compile_definitions(${ogre_next_target} PRIVATE
    OGRE_IGNORE_UNKNOWN_DEBUG
    OGRE_NEXT_VERSION_MAJOR=${OGRE${ogre_next_backward_compatible_suffix}_VERSION_MAJOR}
    OGRE_NEXT_VERSION_MINOR=${OGRE${ogre_next_backward_compatible_suffix}_VERSION_MINOR}
    OGRE_NEXT_VERSION_PATCH=${OGRE${ogre_next_backward_compatible_suffix}_VERSION_PATCH}
    USE_OGRE2_DEFINITIONS=${use_ogre2_definitions}
  )

  # Add OGRE_NEXT_LIBRARY_DIRS to INSTALL_RPATH
  # Append to a copy of CMAKE_INSTALL_RPATH since that is the default
  set(ogre_next_target_install_rpath ${CMAKE_INSTALL_RPATH})
  list(APPEND ogre_next_target_install_rpath ${OGRE_NEXT_LIBRARY_DIRS})
  set_property(
    TARGET ${ogre_next_target}
    PROPERTY INSTALL_RPATH
    ${ogre_next_target_install_rpath}
  )

  target_include_directories(${ogre_next_target}
    PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    # VCT GI needs this
    ${OGRE${ogre_next_backward_compatible_suffix}_INCLUDE}/Hlms/Pbs
    ${OGRE${ogre_next_backward_compatible_suffix}_INCLUDE}/Hlms/Common
  )

  set(gz_ogre_next_target GzOGRENext::GzOGRENext)
  if (NOT GZ_RENDERING_HAVE_OGRE_NEXT)
    set(gz_ogre_next_target GzOGRE2::GzOGRE2)
  endif()
  target_link_libraries(${ogre_next_target}
    PUBLIC
      ${gz-common_LIBRARIES}
      gz-math::eigen3
    PRIVATE
      gz-plugin::register
      ${OPENGL_LIBRARIES}
      terra
      ${gz_ogre_next_target})


  if (TARGET OpenGL::EGL)
    target_link_libraries(${ogre_next_target}
    PRIVATE
        OpenGL::EGL
    )
    add_definitions(-DHAVE_EGL=1)
  endif()

  # You might need to uncomment the following `target_compile_definitions`
  # if you've built Ogre from source in Debug mode
  # https://github.com/OGRECave/ogre-next/blob/003f51a0a90d1cf93fbea3c7302565b07c4f87b0/OgreMain/include/OgrePlatform.h#L350-L372
  #target_compile_definitions(${ogre_next_target} PRIVATE $<$<CONFIG:Debug>:DEBUG=1 _DEBUG=1>)

  # Note that plugins are currently being installed in 2 places: /lib and the engine-plugins dir
  install(TARGETS ${ogre_next_target}
    DESTINATION ${GZ_RENDERING_ENGINE_RELATIVE_INSTALL_DIR}
  )

  if (WIN32)
    # disable MSVC inherit via dominance warning
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /wd4250")
  endif()

  if (NOT (APPLE OR WIN32))
    option(USE_GLX "Use GLX to create OpenGL contexts." ON)
  else()
    option(USE_GLX "Use GLX to create OpenGL contexts." OFF)
  endif()
  if (USE_GLX)
    add_definitions(-DHAVE_GLX=1)
  endif()

  # Build the unit tests
  gz_build_tests(TYPE UNIT
                 SOURCES ${gtest_sources}
                 LIB_DEPS ${ogre_next_target}
                 ENVIRONMENT GZ_RENDERING_INSTALL_PREFIX=${CMAKE_INSTALL_PREFIX})

  install(DIRECTORY "media"
    DESTINATION ${GZ_RENDERING_RELATIVE_RESOURCE_PATH}/ogre_next
  )

endfunction()

if (GZ_RENDERING_HAVE_OGRE_NEXT OR GZ_RENDERING_HAVE_OGRE2)
  gz_add_ogre_next_plugin(false) # add ogre_next plugin
endif()
if (GZ_RENDERING_HAVE_OGRE2)
  gz_add_ogre_next_plugin(true) # add ogre2 plugin
endif()
